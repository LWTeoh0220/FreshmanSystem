## 第二步：唯一的座標系統規則

```
TILE_SIZE = 16  （px，固定，不得更改）
地圖寬 = 160 格
地圖高 = 90 格

格子座標 → 螢幕座標：
  螢幕x = 格子x × 16 - camera.x
  螢幕y = 格子y × 16 - camera.y
  寬度  = 格子w × 16
  高度  = 格子h × 16
```

## 第四步：渲染（render 函數）

只做最基礎的渲染，不做任何紋理，只填純色：

```javascript
const TILE_COLORS = {
  0: '#3a7a20',   // 草地
  1: '#686058',   // 馬路
  2: '#b86858',   // 校內磚紅走道
  3: '#a08060',   // 建築（先統一用這個顏色）
  4: '#2d6818',   // 樹
};

function render(ctx, map, camera) {
  const TS = 16;
  const VIEW_W = 1200;
  const VIEW_H = 620;

  const startCol = Math.floor(camera.x / TS);
  const startRow = Math.floor(camera.y / TS);
  const endCol = Math.min(160, startCol + Math.ceil(VIEW_W / TS) + 1);
  const endRow = Math.min(90,  startRow + Math.ceil(VIEW_H / TS) + 1);

  for (let r = startRow; r < endRow; r++) {
    for (let c = startCol; c < endCol; c++) {
      const tile = map[r][c];
      const px = c * TS - camera.x;
      const py = r * TS - camera.y;
      ctx.fillStyle = TILE_COLORS[tile] || '#3a7a20';
      ctx.fillRect(px, py, TS, TS);
    }
  }
}
```

function drawLabels(ctx, labels, camera) {
  const TS = 16;
  ctx.font = '10px monospace';
  ctx.textAlign = 'center';

  labels.forEach(b => {
    const bx = b.x * TS - camera.x;
    const by = b.y * TS - camera.y;
    const bw = b.w * TS;
    const bh = b.h * TS;

    if (bx + bw < 0 || bx > 1200 || by + bh < 0 || by > 620) return;

    const cx = bx + bw / 2;
    const cy = by + bh / 2;
    const lw = ctx.measureText(b.name).width + 8;

    ctx.fillStyle = 'rgba(244,232,209,0.90)';
    ctx.fillRect(cx - lw/2, cy - 8, lw, 16);
    ctx.fillStyle = '#3a2a18';
    ctx.fillText(b.name, cx, cy + 4);
  });
}

## 第六步：WASD 移動與相機

```javascript
const player = { px: 13 * 16, py: 41 * 16 };  // 初始位置：小道1與主道1交叉點
const camera = { x: 0, y: 0 };
const keys = {};

function updateCamera() {
  camera.x = Math.max(0, Math.min(160*16 - VIEW_W, player.px - VIEW_W/2));
  camera.y = Math.max(0, Math.min(90*16  - VIEW_H, player.py - VIEW_H/2));
}

function isWalkable(map, px, py) {
  const col = Math.floor(px / 16);
  const row = Math.floor(py / 16);
  if (col < 0 || row < 0 || col >= 160 || row >= 90) return false;
  return map[row][col] !== 3 && map[row][col] !== 4;
}
```
